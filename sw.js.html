const CACHE_NAME = 'jb-erp-v3.1'; const urlsToCache = [ './', './index.html', './manifest.json' ]; // Instala o Service Worker e guarda os arquivos básicos no cache do celular self.addEventListener('install', event => { event.waitUntil( caches.open(CACHE_NAME) .then(cache => cache.addAll(urlsToCache)) ); }); // Intercepta as requisições para carregar o app mais rápido self.addEventListener('fetch', event => { event.respondWith( caches.match(event.request) .then(response => response || fetch(event.request)) ); });
